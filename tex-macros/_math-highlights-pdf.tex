\newcommand{\mathhlbox}[3]{%
  \mathchoice
    {\begingroup\fboxsep=2pt\fboxrule=0.4pt\fcolorbox{#2}{#1}{$\displaystyle #3$}\endgroup}
    {\begingroup\fboxsep=2pt\fboxrule=0.4pt\fcolorbox{#2}{#1}{$\textstyle #3$}\endgroup}
    {\begingroup\fboxsep=1.4pt\fboxrule=0.35pt\fcolorbox{#2}{#1}{$\scriptstyle #3$}\endgroup}
    {\begingroup\fboxsep=1pt\fboxrule=0.3pt\fcolorbox{#2}{#1}{$\scriptscriptstyle #3$}\endgroup}%
}
